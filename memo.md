# ====== Python開発メモ ======

## 初期セットアップ
# 1.仮想環境を作成
python -m venv .venv
# 2.requirements.txt の内容をインストール(仮想環境内で行う)
pip install -r requirements.txt


# 仮想環境(.venv)の起動方法（PowerShell）
cd C:\path\to\your\project
.\.venv\Scripts\Activate.ps1

# Pythonファイルの実行
python main.py

# 仮想環境の終了
deactivate
